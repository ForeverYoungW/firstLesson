WITH X AS
(
  SELECT DY,
         DAY(DY) AS DM,
         DATEPART(M, DY) AS MTH,
         DATEPART(DW, DY) AS DW,
         DATEPART(WW, DY) AS WK
    FROM (SELECT DATEADD(DAY, -DAY(GETDATE()) + 1, GETDATE()) AS DY) AS TMP1
   UNION ALL
  SELECT DATEADD(D, 1, DY),
         DAY(DATEADD(D, 1, DY)),
         MTH,
         DATEPART(DW, DATEADD(D, 1, DY)),
         DATEPART(WK, DATEADD(D, 1, DY))
    FROM X
   WHERE DATEPART(M, DATEADD(D, 1, DY)) = MTH
)
SELECT MAX(CASE DW WHEN 1 THEN DM END) AS Su,
       MAX(CASE DW WHEN 2 THEN DM END) AS Mo,
       MAX(CASE DW WHEN 3 THEN DM END) AS Tu,
       MAX(CASE DW WHEN 4 THEN DM END) AS We,
       MAX(CASE DW WHEN 5 THEN DM END) AS Th,
       MAX(CASE DW WHEN 6 THEN DM END) AS Fr,
       MAX(CASE DW WHEN 7 THEN DM END) AS Sa
  FROM X
 GROUP BY WK
 ORDER BY WK
