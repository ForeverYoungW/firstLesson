(
  SELECT DY, YEAR(DY) YR
    FROM (SELECT GETDATE() - DATEPART(DY, GETDATE()) + 1 AS DY
            FROM T1
         ) AS TMP1
   UNION ALL
  SELECT DATEADD(DAY, 1, DY), YR
    FROM X
   WHERE YEAR(DATEADD(DAY, 1, DY)) = YR
)
