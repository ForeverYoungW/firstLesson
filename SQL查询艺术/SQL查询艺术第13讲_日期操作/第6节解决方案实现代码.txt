WITH X AS
(
  SELECT DY, MONTH(DY) AS MTH,
         CASE
           WHEN DATEPART(DW, DY) = 2 THEN 1
           ELSE 0
         END AS IS_MONDAY
    FROM (SELECT DATEADD(DAY, -DAY(GETDATE()) + 1, GETDATE()) AS DY) AS TMP1
   UNION ALL
  SELECT DATEADD(DAY, 1, DY),MTH,
         CASE
           WHEN DATEPART(DW, DATEADD(DAY, 1, DY)) = 2 THEN 1
           ELSE 0
         END
    FROM X
   WHERE MONTH(DATEADD(DAY, 1, DY)) = MTH
)
SELECT MIN(DY) AS FIRST_MONDAY,
       MAX(DY) AS LAST_MONDAY
  FROM X
 WHERE IS_MONDAY = 1
