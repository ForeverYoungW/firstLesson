
SELECT DNAME
  FROM (SELECT DNAME,
               ITER.POS,
               SUBSTRING(DNAME, ITER.POS, 1) AS C,
               ASCII(SUBSTRING(DNAME, ITER.POS, 1)) AS CODE
          FROM DEPT_TEST
         CROSS JOIN (SELECT ID AS POS
                       FROM T100
                    ) ITER
         WHERE ITER.POS <= LEN(DNAME)
       ) AS X
 WHERE CODE BETWEEN 48 AND 57
    OR CODE BETWEEN 65 AND 90
    OR CODE BETWEEN 97 AND 122
 GROUP BY DNAME
HAVING COUNT(DNAME) = LEN(DNAME)
