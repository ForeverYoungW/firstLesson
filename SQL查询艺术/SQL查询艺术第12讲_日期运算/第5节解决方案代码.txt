SELECT DATEDIFF(DAY, ALLEN_HD, WARD_HD) * 24 AS HR,
       DATEDIFF(DAY, ALLEN_HD, WARD_HD) * 24 * 60 AS MIN,
       DATEDIFF(DAY, ALLEN_HD, WARD_HD) * 24 * 60 * 60 AS SEC
  FROM (SELECT MAX(CASE WHEN ENAME = 'WARD' THEN HIREDATE END) AS WARD_HD,
               MAX(CASE WHEN ENAME = 'ALLEN' THEN HIREDATE END) AS ALLEN_HD
          FROM EMP
       ) X
