WITH EmpsCTE AS
(
  SELECT EMPNO, MGR, ENAME, 0 AS LEVEL
    FROM EMP
   WHERE EMPNO = 7839
   UNION ALL
  SELECT E.EMPNO, E.MGR, E.ENAME, P.LEVEL + 1 AS LEVEL
    FROM EmpsCTE AS P
    JOIN EMP AS E ON E.MGR = P.EMPNO
)
SELECT E1.ENAME AS Employee,
       E2.ENAME AS Boss,
       E1.EMPNO,
       E1.MGR
  FROM EmpsCTE AS E1
 INNER JOIN EMP AS E2 ON E1.MGR = E2.EMPNO
