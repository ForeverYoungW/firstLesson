SELECT EMP.DEPTNO,
       DNAME,
       SUM(COMM) AS SUM,
       COUNT(EMPNO) AS COUNT,
       AVG(COALESCE(COMM, 0)) AS AVG
  FROM EMP
 INNER JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO
 WHERE EMP.DEPTNO IN (20, 30)
 GROUP BY EMP.DEPTNO, DNAME

2:
SELECT DEPTNO, AVG(SAL) AS AVG_SAL
  FROM EMP
 GROUP BY DEPTNO
HAVING AVG(SAL) > (SELECT AVG(SAL)
                     FROM EMP
                  )

3:
SELECT EMP.EMPNO, ENAME, JOB, MGR, HIREDATE, SAL, COMM, DEPTNO
  FROM EMP
 INNER JOIN (SELECT MAX(EMPNO) AS EMPNO
               FROM EMP
            ) M ON EMP.EMPNO = M.EMPNO
4:
SELECT EMP.EMPNO, EMP.ENAME, EMP.JOB, EMP.MGR, EMP.HIREDATE, EMP.SAL, EMP.COMM, EMP.DEPTNO
  FROM EMP
 INNER JOIN (SELECT MIN(DEPTNO) AS DEPTNO
               FROM DEPT
            ) M ON EMP.DEPTNO = M.DEPTNO
