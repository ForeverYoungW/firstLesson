SELECT E1.ENAME,
       E1.SAL,
       (SELECT SUM(E2.SAL)
          FROM EMP E2
         WHERE E2.EMPNO <= E1.EMPNO
       ) AS TOTAL
  FROM EMP AS E1
 WHERE E1.EMPNO <= 7566
 ORDER BY TOTAL;

2:
SELECT E1.ENAME,
       E1.SAL,
       (SELECT SUM(E2.SAL)
          FROM EMP E2
         WHERE E2.EMPNO <= E1.EMPNO
           AND E2.DEPTNO = 10
       ) AS TOTAL
  FROM EMP AS E1
 WHERE E1.DEPTNO = 10
 ORDER BY TOTAL;
